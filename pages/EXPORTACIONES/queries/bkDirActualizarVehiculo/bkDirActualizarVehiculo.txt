with datos_anteriores as (
  select to_jsonb(v) as data
  from public.vehiculos v
  where v.id = '{{ appsmith.store.cp_update_context?.entity_id }}'::uuid
),

actualizar as (
  update public.vehiculos
  set
    tipo_vehiculo_terrestre_id = nullif('{{ slDirVehTipoVehiculo.selectedOptionValue || "" }}', '')::uuid,
    marca = upper(trim('{{ slDirVehMarca.selectedOptionValue || "" }}')),
    modelo = upper(trim('{{ inpDirVehModelo.text || "" }}')),
    color = upper(trim('{{ slDirVehColor.selectedOptionValue || "" }}')),
    capacidad_carga_kg = nullif('{{ inpDirVehCapacidadKg.text || "" }}', '')::numeric,
    soat_no = upper(trim('{{ inpDirVehSoatNo.text || "" }}')),
    soat_vencimiento = nullif('{{ dpDirVehSoatVencimiento.selectedDate || "" }}', '')::date,
    updated_by = '{{ appsmith.store.user_id }}'::uuid,
    updated_at = now()
  where id = '{{ appsmith.store.cp_update_context?.entity_id }}'::uuid
  returning *
),

auditoria as (
  insert into public.auditoria_entidades (
    tenant_id,
    modulo,
    entidad,
    entidad_id,
    operacion,
    motivo,
    datos_anteriores,
    datos_nuevos,
    usuario_id,
    autorizado_por
  )
  select
    '{{ appsmith.store.cp_tenant_id }}'::uuid,
    'CARGUE_PLANTA',
    'VEHICULOS',
    '{{ appsmith.store.cp_update_context?.entity_id }}'::uuid,
    'DIRECTORIO_UPDATE',
    upper(trim('{{ inpDirUpdateMotivo.text || "" }}')),
    da.data,
    to_jsonb(a),
    '{{ appsmith.store.user_id }}'::uuid,
    '{{ appsmith.store.user_id }}'::uuid
  from actualizar a
  cross join datos_anteriores da
  returning id
)

select
  a.id,
  a.placa,
  a.marca,
  a.modelo,
  a.color,
  a.capacidad_carga_kg,
  a.soat_no,
  a.soat_vencimiento
from actualizar a;
